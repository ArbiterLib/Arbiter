110 /**
111  * Represents a resolved dependency graph, preserving relationships between
112  * dependencies.
113  *
114  * Resolved graphs are ordered in such a way that projects without any
115  * dependencies all exist at depth 0, projects which depend only upon those
116  * exist at depth 1, projects which depend upon depth 0 or 1 exist at depth 2,
117  * etc.
118  *
119  * In other words, the graph can be thought of as being "install-ordered," where
120  * the projects listed within one depth index may be safely installed in
121  * parallel with respect to each another, and the projects within _each
122  * successive depth_ must be installed only after the projects in _all previous
123  * depths_ have been completely installed.
124  */
125 typedef struct ArbiterResolvedDependencyGraph ArbiterResolvedDependencyGraph;
